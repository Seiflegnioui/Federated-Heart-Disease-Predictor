\section{Contexte \& Objectifs}
% ============================================================================

\begin{frame}{Le dilemme des donn\'{e}es m\'{e}dicales}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \begin{block}{Le probl\`{e}me de la centralisation}
        \begin{itemize}
          \item Les mod\`{e}les d'IA n\'{e}cessitent beaucoup de donn\'{e}es pour \^{e}tre performants.
          \item Les donn\'{e}es m\'{e}dicales sont sensibles, isol\'{e}es dans des h\^{o}pitaux diff\'{e}rents (silos).
          \item Le transfert de ces donn\'{e}es est bloqu\'{e} par la r\'{e}glementation (RGPD).
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{exampleblock}{La solution : L'Apprentissage F\'{e}d\'{e}r\'{e}}
        Au lieu d'amener les donn\'{e}es au mod\`{e}le (centralisation), on \textbf{am\`{e}ne le mod\`{e}le aux donn\'{e}es}.
        Seuls les \emph{poids} du mod\`{e}le sont \'{e}chang\'{e}s, garantissant la confidentialit\'{e}.
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Objectifs du projet (Option 4)}
  \begin{alertblock}{Cahier des charges respect\'{e}}
    \begin{enumerate}
      \item \textbf{Domaine :} Simulation m\'{e}dicale sur donn\'{e}es tabulaires.
      \item \textbf{R\'{e}seau :} Perceptron Multicouche (MLP).
      \item \textbf{F\'{e}d\'{e}ration :} Simulation d'au moins 8 clients (h\^{o}pitaux).
      \item \textbf{Comparaison algorithmique :} \'{E}valuation de 4 algorithmes sp\'{e}cifiques.
      \item \textbf{H\'{e}t\'{e}rog\'{e}n\'{e}it\'{e} :} \'{E}tude des sc\'{e}narios IID (donn\'{e}es \'{e}quilibr\'{e}es) et Non-IID (donn\'{e}es asym\'{e}triques).
    \end{enumerate}
  \end{alertblock}
\end{frame}

\section{Architecture \& M\'{e}thodologie}
% ============================================================================

\begin{frame}{Architecture de la F\'{e}d\'{e}ration Hospitali\`{e}re}
  \begin{columns}[c]
    \begin{column}{0.45\textwidth}
      \begin{itemize}
        \item \textbf{Topologie en \'{e}toile :} 1 serveur global coordonne 8 h\^{o}pitaux locaux.
        \item \textbf{Processus (1 Round) :}
        \begin{enumerate}
            \item Le serveur envoie le mod\`{e}le aux h\^{o}pitaux.
            \item Chaque h\^{o}pital l'entra\^{i}ne sur ses propres patients.
            \item Les h\^{o}pitaux renvoient leurs mises \`{a} jour.
            \item Le serveur fusionne tout.
        \end{enumerate}
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{figure}
        \centering
        \resizebox{0.9\textwidth}{!}{%
        \begin{tikzpicture}[
            server/.style={circle, draw=mDarkTeal, fill=mDarkTeal!20, thick, minimum size=1.8cm, align=center, font=\bfseries},
            client/.style={rectangle, rounded corners, draw=mLightBrown, fill=mLightBrown!20, thick, minimum size=1cm, align=center},
            arrow/.style={<->, >=Stealth, thick, mDarkTeal!70}
          ]
          \node[server] (S) {Serveur\\Global};

             \node[client] (H\i) at (\a:2.8cm) {H\^{o}p \i};
             \draw[arrow] (S) -- (H\i);
          }
        \end{tikzpicture}
        }
      \end{figure}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Donn\'{e}es et Mod\`{e}le}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Dataset : Heart Disease UCI}
        \begin{itemize}
          \item \textbf{Taille :} 1\,025 patients.
          \item \textbf{Attributs :} 13 indicateurs (âge, pression art\'{e}rielle, cholest\'{e}rol, ECG, etc.).
          \item \textbf{Cible :} Binaire (Malade / Sain).
          \item \textbf{Pr\'{e}traitement :} Encodage One-Hot et Standardisation Z-Score (23 features finales).
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Mod\`{e}le : MLP L\'{e}ger}
        \begin{itemize}
          \item \textbf{Architecture :} Entr\'{e}e(23) $\to$ Couche 1(64) $\to$ Couche 2(32) $\to$ Sortie(1).
          \item \textbf{Taille :} $\sim$2\,600 param\`{e}tres.
          \item \textbf{Poids :} Moins de 11 Ko ! Tr\`{e}s facile \`{a} transf\'{e}rer sur le r\'{e}seau.
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{La R\'{e}partition des Donn\'{e}es (IID vs Non-IID)}
  Dans le monde r\'{e}el, un h\^{o}pital cardiologique (A) a beaucoup de patients malades, tandis qu'une clinique g\'{e}n\'{e}rale (B) a surtout des patients sains.
  \vspace{0.3cm}
  
  \begin{itemize}
      \item \textbf{Sc\'{e}nario IID (Id\'{e}al) :} Les donn\'{e}es sont m\'{e}lang\'{e}es et r\'{e}parties al\'{e}atoirement de mani\`{e}re \'{e}gale entre les 8 h\^{o}pitaux.
      \item \textbf{Sc\'{e}nario Non-IID (R\'{e}aliste) :} R\'{e}partition in\'{e}gale des classes simul\'{e}e via une \emph{distribution de Dirichlet} ($\alpha=0.5$). Certains h\^{o}pitaux se sp\'{e}cialisent involontairement.
  \end{itemize}

  \begin{alertblock}{Le d\'{e}fi du Non-IID}
      L'h\'{e}t\'{e}rog\'{e}n\'{e}it\'{e} cr\'{e}e une "d\'{e}rive" (Client Drift) : chaque h\^{o}pital optimise son mod\`{e}le dans une direction diff\'{e}rente. Le serveur doit corriger cela.
  \end{alertblock}
\end{frame}

\section{Algorithmes & R\'{e}sultats}
% ============================================================================

\begin{frame}{Les 4 Algorithmes \'{E}valu\'{e}s}
  Nous avons impl\'{e}ment\'{e} et compar\'{e} 4 strat\'{e}gies :
  \vspace{0.3cm}
  \begin{enumerate}
      \item \textbf{Local SGD (Sans collaboration) :} Chaque h\^{o}pital s'entra\^{i}ne seul. Le pire sc\'{e}nario en termes de performances globales.
      \item \textbf{FedAvg (Le Standard) :} Le serveur fait une simple moyenne pond\'{e}r\'{e}e des param\`{e}tres de tous les h\^{o}pitaux.
      \item \textbf{FedAdam (L'Adaptatif) :} Le serveur utilise une m\'{e}moire (inertie/momentum) pour lisser les mises \`{a} jour et acc\'{e}l\'{e}rer la convergence.
      \item \textbf{FedProx (Le R\'{e}gulateur) :} Ajoute une contrainte locale (terme proximal) pour emp\^{e}cher les h\^{o}pitaux de trop s'\'{e}loigner du mod\`{e}le global, excellent pour le Non-IID.
  \end{enumerate}
\end{frame}

\begin{frame}{R\'{e}sultats : Performance Globale (Accuracy)}
  \begin{center}
    \begin{tabular}{lcc}
      \toprule
      \textbf{Algorithme} & \textbf{Pr\'{e}cision IID} & \textbf{Pr\'{e}cision Non-IID} \\
      \midrule
      \textbf{Local SGD}  & 81.3\% & \textcolor{mRed}{62.1\%} \\
      \textbf{FedAvg}     & 91.8\% & 91.3\% \\
      \textbf{FedAdam}    & \textcolor{mLightGreen}{92.4\%} & 90.7\% \\
      \textbf{FedProx}    & 91.8\% & \textcolor{mLightGreen}{91.7\%} \\
      \bottomrule
    \end{tabular}
  \end{center}
  
  \vspace{0.3cm}
  \begin{itemize}
      \item L'absence de collaboration (\textbf{Local SGD}) est catastrophique sur des donn\'{e}es asym\'{e}triques.
      \item \textbf{FedAdam} est le plus performant en situation id\'{e}ale (IID).
      \item \textbf{FedProx} r\'{e}siste le mieux \`{a} l'h\'{e}t\'{e}rog\'{e}n\'{e}it\'{e} (Non-IID) gr\^{a}ce \`{a} sa r\'{e}gularisation.
  \end{itemize}
\end{frame}

\begin{frame}{Co\^{u}t de Communication \& Efficacit\'{e}}
  \begin{block}{Un syst\`{e}me extr\^{e}mement l\'{e}ger}
    \begin{itemize}
        \item Notre mod\`{e}le MLP p\`{e}se seulement \textbf{$\sim$10 Ko}.
        \item Lors d'un round, 8 h\^{o}pitaux re\c{c}oivent et envoient le mod\`{e}le $\to$ $\sim$160 Ko par round au total.
        \item Pour la totalit\'{e} des 15 rounds d'entra\^{i}nement, le co\^{u}t r\'{e}seau est d'\textbf{environ 2.5 Mo au total}.
    \end{itemize}
  \end{block}
  
  \vspace{0.2cm}
  \begin{exampleblock}{Conclusion Pratique}
      Un tel syst\`{e}me peut \^{e}tre d\'{e}ploy\'{e} imm\'{e}diatement m\^{e}me sur des connexions hospitali\`{e}res tr\`{e}s lentes, garantissant un apprentissage rapide et peu co\^{u}teux sans bouger la moindre donn\'{e}e patient.
  \end{exampleblock}
\end{frame}

\section{Conclusion}
% ============================================================================

\begin{frame}{Conclusion}
  \begin{itemize}
    \item \textbf{Validation de l'Apprentissage F\'{e}d\'{e}r\'{e} :} Nous avons prouv\'{e} qu'une collaboration entre h\^{o}pitaux, sans jamais centraliser les donn\'{e}es, permet d'atteindre plus de 91\% de pr\'{e}cision sur la d\'{e}tection de maladies cardiaques.
    \item \textbf{Choix Algorithmique :} 
    \begin{itemize}
        \item \emph{FedAvg} est un excellent standard par d\'{e}faut.
        \item \emph{FedProx} doit \^{e}tre privil\'{e}gi\'{e} lorsque les h\^{o}pitaux ont des profils de patients tr\`{e}s diff\'{e}rents (Non-IID).
    \end{itemize}
    \item \textbf{Perspectives :} Ajouter du chiffrement homomorphe pour emp\^{e}cher l'analyse des poids (Gradient Inversion Attacks) et l'\'{e}tendre \`{a} des donn\'{e}es d'imagerie m\'{e}dicale (CNN).
  \end{itemize}
\end{frame}
