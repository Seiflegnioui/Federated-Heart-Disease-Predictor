% ============================================================================
\begin{frame}{Introduction --- Probl\'{e}matique}
  \begin{block}{Limites de l'apprentissage centralis\'{e}}
    \begin{enumerate}
      \item \textbf{Vie priv\'{e}e} --- Les patients ne consentent pas au transfert de leurs dossiers m\'{e}dicaux vers un serveur central
      \item \textbf{R\'{e}glementation} --- Les lois imposent la localisation des donn\'{e}es au sein de l'\'{e}tablissement
      \item \textbf{Volume} --- Le transfert de larges volumes de donn\'{e}es m\'{e}dicales est co\^{u}teux et lent
      \item \textbf{H\'{e}t\'{e}rog\'{e}n\'{e}it\'{e}} --- Chaque h\^{o}pital a une population diff\'{e}rente (biais d\'{e}mographique, protocoles)
    \end{enumerate}
  \end{block}
  \vfill
  \begin{exampleblock}{Cons\'{e}quence directe}
    Les donn\'{e}es restent \textbf{fragment\'{e}es} en silos isol\'{e}s $\longrightarrow$ les mod\`{e}les locaux souffrent d'un manque de g\'{e}n\'{e}ralisation.
  \end{exampleblock}
\end{frame}

% ============================================================================
% SLIDE 5 : L'Apprentissage F\'{e}d\'{e}r\'{e}
% ============================================================================
\begin{frame}{Introduction --- L'Apprentissage F\'{e}d\'{e}r\'{e}}
  \begin{block}{D\'{e}finition (McMahan \textit{et al.}, 2017)}
    L'apprentissage f\'{e}d\'{e}r\'{e} (\textit{Federated Learning}) est un paradigme d'entra\^{i}nement collaboratif o\`{u} le \textbf{mod\`{e}le voyage vers les donn\'{e}es}, et non l'inverse.
  \end{block}
  \vspace{0.3cm}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Principe g\'{e}n\'{e}ral :}
      \begin{enumerate}
        \item Le serveur envoie le mod\`{e}le global $w^{(t)}$
        \item Chaque client $k$ entra\^{i}ne localement
        \item Les clients renvoient leurs mises \`{a} jour $w_k^{(t+1)}$
        \item Le serveur agr\`{e}ge les poids
        \item R\'{e}p\'{e}ter pendant $T$ rounds
      \end{enumerate}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Avantages cl\'{e}s :}
      \begin{itemize}
        \item[\checkmark] Donn\'{e}es \textbf{jamais} partag\'{e}es
        \item[\checkmark] Conformit\'{e} r\'{e}glementaire
        \item[\checkmark] Exploitation de donn\'{e}es distribu\'{e}es
        \item[\checkmark] R\'{e}duction des co\^{u}ts de transfert
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\section{Algorithmes d'Optimisation}

% ============================================================================
% SLIDE 12 : Local SGD
% ============================================================================
\begin{frame}{Algorithme 1 --- Local SGD (Baseline)}
  \begin{block}{Principe}
    Chaque h\^{o}pital entra\^{i}ne son mod\`{e}le \textbf{ind\'{e}pendamment}, sans aucune communication avec le serveur.
  \end{block}
  \vspace{0.3cm}
  \begin{columns}[T]
    \begin{column}{0.5\textwidth}
      \textbf{Mise \`{a} jour locale :}
      \[
        w_k^{(t+1)} = w_k^{(t)} - \eta \nabla F_k(w_k^{(t)})
      \]
      o\`{u} $F_k$ est la perte locale du client $k$.
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{alertblock}{Limitations}
        \begin{itemize}
          \item Aucune agr\'{e}gation
          \item Pas de g\'{e}n\'{e}ralisation inter-sites
          \item Fortement d\'{e}pendant des donn\'{e}es locales
        \end{itemize}
      \end{alertblock}
    \end{column}
  \end{columns}
  \vfill
  \centering
  \textcolor{accentblue}{\textit{$\Rightarrow$ Sert de r\'{e}f\'{e}rence (baseline) pour \'{e}valuer l'apport de la f\'{e}d\'{e}ration.}}
\end{frame}

% ============================================================================
% SLIDE 13 : FedAvg
% ============================================================================
\begin{frame}{Algorithme 2 --- FedAvg}
  \begin{block}{Federated Averaging (McMahan \textit{et al.}, 2017)}
    Agr\'{e}gation par \textbf{moyenne pond\'{e}r\'{e}e} des poids locaux, proportionnellement \`{a} la taille du dataset de chaque client.
  \end{block}
  \vspace{0.3cm}
  \begin{exampleblock}{Formule d'agr\'{e}gation}
    \[
      w^{(t+1)} = \sum_{k=1}^{K} \frac{n_k}{n} \, w_k^{(t+1)}
    \]
    o\`{u} $n_k = |\mathcal{D}_k|$ est le nombre d'\'{e}chantillons du client $k$, et $n = \sum_{k} n_k$.
  \end{exampleblock}
  \vspace{0.3cm}
  \begin{itemize}
    \item \textbf{Entra\^{i}nement local} : $E = 5$ \'{e}poques par round
    \item \textbf{Rounds de communication} : $T = 15$
    \item \textbf{Clients participants} : $K = 8$ (tous s\'{e}lectionn\'{e}s)
  \end{itemize}
\end{frame}

% ============================================================================
% SLIDE 14 : FedAdam
% ============================================================================
\begin{frame}{Algorithme 3 --- FedAdam}
  \begin{block}{Optimisation adaptative c\^{o}t\'{e} serveur}
    Le serveur applique l'optimiseur \textbf{Adam} sur le pseudo-gradient agr\'{e}g\'{e}, au lieu d'une simple moyenne.
  \end{block}
  \vspace{0.2cm}
  \begin{exampleblock}{\'{E}quations de mise \`{a} jour}
    \textbf{Pseudo-gradient :}
    \[
      \Delta^{(t)} = \sum_{k=1}^{K} \frac{n_k}{n}\left(w_k^{(t+1)} - w^{(t)}\right)
    \]
    \textbf{Moments adaptatifs (serveur) :}
    \[
      m^{(t+1)} = \beta_1 \, m^{(t)} + (1 - \beta_1)\, \Delta^{(t)}
    \]
    \[
      v^{(t+1)} = \beta_2 \, v^{(t)} + (1 - \beta_2)\, (\Delta^{(t)})^2
    \]
    \[
      w^{(t+1)} = w^{(t)} + \eta_s \frac{m^{(t+1)}}{\sqrt{v^{(t+1)}} + \epsilon}
    \]
  \end{exampleblock}
\end{frame}

% ============================================================================
% SLIDE 15 : FedProx
% ============================================================================
\begin{frame}{Algorithme 4 --- FedProx}
  \begin{block}{Terme proximal pour la r\'{e}gularisation (Li \textit{et al.}, 2020)}
    FedProx ajoute un \textbf{terme de p\'{e}nalit\'{e} proximale} \`{a} la fonction de perte locale pour limiter la \textbf{d\'{e}rive des clients}.
  \end{block}
  \vspace{0.3cm}
  \begin{exampleblock}{Fonction objectif modifi\'{e}e}
    \[
      \min_{w_k} \; F_k(w_k) + \frac{\mu}{2} \left\| w_k - w^{(t)} \right\|^2
    \]
    avec $\mu = 0.1$ (coefficient proximal).
  \end{exampleblock}
  \vspace{0.3cm}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Avantages :}
      \begin{itemize}
        \item[\checkmark] Stabilit\'{e} en environnement Non-IID
        \item[\checkmark] Contr\^{o}le de la divergence locale
        \item[\checkmark] Simple \`{a} impl\'{e}menter
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Intuition :}
      \begin{itemize}
        \item Le terme $\frac{\mu}{2}\|w_k - w^{(t)}\|^2$ emp\^{e}che le mod\`{e}le local de trop s'\'{e}loigner du mod\`{e}le global
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{R\'{e}sultats --- Divergence Locale}
  \begin{columns}[c]
    \begin{column}{0.5\textwidth}
      \begin{figure}
        \centering
        \includegraphics[width=\textwidth]{example-image}
        \caption{Norme L2 de la divergence locale par round}
      \end{figure}
    \end{column}
    \begin{column}{0.47\textwidth}
      \begin{block}{Mesure de divergence}
        \[
          D_k^{(t)} = \left\| w_k^{(t)} - w^{(t)} \right\|_2
        \]
        Moyenne sur les $K$ clients :
        \[
          \bar{D}^{(t)} = \frac{1}{K}\sum_{k=1}^{K} D_k^{(t)}
        \]
      \end{block}
      \vspace{0.2cm}
      \begin{alertblock}{R\'{e}sultats}
        \begin{itemize}
          \item FedProx : divergence \textbf{la plus faible}
          \item FedAvg : divergence mod\'{e}r\'{e}e
          \item Local SGD : divergence \textbf{croissante} (pas de correction)
        \end{itemize}
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

% ============================================================================
\begin{frame}{R\'{e}sultats --- Co\^{u}t de Communication}
  \begin{block}{Calcul du co\^{u}t total de communication}
    \[
      C_{\text{total}} = 2 \times K \times T \times S_{\text{mod\`{e}le}}
    \]
    o\`{u} le facteur $2$ correspond \`{a} l'envoi (serveur $\to$ client) et au retour (client $\to$ serveur).
  \end{block}
  \vspace{0.2cm}
  \centering
  \begin{tabular}{ll}
    \toprule
    \textbf{Param\`{e}tre} & \textbf{Valeur} \\
    \midrule
    Taille du mod\`{e}le ($S_{\text{mod\`{e}le}}$) & $\sim$10\,KB (2\,600 param. $\times$ 4 octets) \\
    Clients ($K$) & 8 \\
    Rounds ($T$) & 15 \\
    \midrule
    \textbf{Co\^{u}t total} & $2 \times 8 \times 15 \times 10 = $ \textbf{2\,400\,KB $\approx$ 2.5\,MB} \\
    \bottomrule
  \end{tabular}
  \vfill
  \begin{exampleblock}{Ratio performance / communication}
    \centering
    FedProx atteint $\sim$91\% d'accuracy pour seulement \textbf{2.5\,MB} de communication totale.\\
    $\Rightarrow$ Extr\^{e}mement efficient gr\^{a}ce \`{a} la compacit\'{e} du MLP.
  \end{exampleblock}
\end{frame}

          \item \textbf{Mod\`{e}le simple} : MLP compact, non repr\'{e}sentatif de cas complexes
          \item \textbf{Donn\'{e}es tabulaires uniquement} : pas d'imagerie m\'{e}dicale
        \end{enumerate}
      \end{alertblock}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{exampleblock}{Perspectives d'am\'{e}lioration}
        \begin{enumerate}
          \item \textbf{Confidentialit\'{e} diff\'{e}rentielle} : int\'{e}grer du bruit calibr\'{e} ($\epsilon$-DP)
          \item \textbf{Agr\'{e}gation s\'{e}curis\'{e}e} : protocoles de calcul multipartite
          \item \textbf{Passage \`{a} l'\'{e}chelle} : 50+ clients, datasets plus grands
          \item \textbf{D\'{e}ploiement r\'{e}el} : int\'{e}gration avec des syst\`{e}mes hospitaliers (HL7/FHIR)
          \item \textbf{Architectures avanc\'{e}es} : CNNs pour l'imagerie, Transformers
        \end{enumerate}

\begin{frame}{Conclusion}
  \begin{block}{Quatre enseignements cl\'{e}s}
    \begin{enumerate}
      \item \textbf{L'apprentissage f\'{e}d\'{e}r\'{e} est viable} pour la pr\'{e}diction de maladies cardiaques, m\^{e}me avec un mod\`{e}le compact ($\sim$2\,600 param\`{e}tres) et un co\^{u}t de communication minimal ($\sim$2.5\,MB)

      \item \textbf{FedAdam acc\'{e}l\`{e}re la convergence} gr\^{a}ce \`{a} l'optimisation adaptative c\^{o}t\'{e} serveur, offrant le meilleur compromis vitesse/performance en IID
    \end{enumerate}
